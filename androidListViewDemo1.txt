        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_list_view_demo);


     CustomAdapter adapter=new CustomAdapter(this,mobileArray,prgmImages);
        listView =  findViewById(R.id.list_item);

                Toast.makeText(ListViewDemoActivity.this,"U Clicked:"+ mobileArray[position],Toast.LENGTH_SHORT).show();

            }
        });


    }
}





//CustomAdaptor.java



package com.example.ashwinipalve.testapp.activity;

import android.app.Activity;
import android.support.annotation.NonNull;
import android.support.annotation.Nullable;
import android.util.Base64;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ArrayAdapter;
import android.widget.BaseAdapter;
import android.widget.ImageView;
import android.widget.TextView;

import com.example.ashwinipalve.testapp.R;

/**
 * Created by Ashwini Palve on 15/01/2018.
 */

public class CustomAdapter extends BaseAdapter
{   private final Activity context;
    private final String[] mobileArray;
     private final int[]  prgmImages;

    public CustomAdapter(Activity context, String[] mobileArray, int[] prgmImages) {
        this.context = context;
        this.mobileArray = mobileArray;
        this.prgmImages = prgmImages;
    }


    @Override
    public int getCount() {
        return mobileArray.length;
    }


    @Override
    public Object getItem(int position) {
        return position;
    }

    @Override
    public long getItemId(int position) {
        return position;
    }

    @NonNull
    @Override
    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {



        LayoutInflater inflater=LayoutInflater.from(context);
        View rowView=inflater.inflate(R.layout.progrmlist,null);

        //to set the text and image into the inflater

        TextView textView=rowView.findViewById(R.id.textView1);
        ImageView imageView=rowView.findViewById(R.id.imageView1);

        textView.setText(mobileArray[position]);
        imageView.setImageResource(prgmImages[position]);

        return rowView;
    }
}
