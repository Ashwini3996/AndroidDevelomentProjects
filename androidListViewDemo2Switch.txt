package com.example.ashwinipalve.testapp.activity;

 */

public class MyListViewAdapter extends BaseAdapter
{

    private final Activity context;
   /* private final String[] mArray;
    private final int[]  mImages;*/

   ArrayList<MyListViewPojo> arrayList;


    public MyListViewAdapter(Activity context,ArrayList<MyListViewPojo> arrayList )
    {
        this.context = context;
        this.arrayList = arrayList;

    }


    @Override
    public int getCount() {
        return arrayList.size();
    }


    @Override
    public Object getItem(int position) {
        return position;
    }


    @Override
    public long getItemId(int position) {
        return position;

    }


    @Override
    public View getView(int position, View view, ViewGroup viewGroup) {

        LayoutInflater inflater=LayoutInflater.from(context);
        View rowView=inflater.inflate(R.layout.programlist1,null);
        TextView textView=rowView.findViewById(R.id.textView1);
        ImageView imageView=rowView.findViewById(R.id.imageView1);

        final Switch simpleswitch=rowView.findViewById(R.id.switch1);
        simpleswitch.setOnCheckedChangeListener(new CompoundButton.OnCheckedChangeListener() {
            String statusSwitch;
            @Override
            public void onCheckedChanged(CompoundButton compoundButton, boolean b) {
                if (simpleswitch.isChecked())
                    statusSwitch =simpleswitch.getTextOn().toString();
                else
                    statusSwitch = simpleswitch.getTextOff().toString();
                Toast.makeText(context, "Switch :" + statusSwitch , Toast.LENGTH_LONG).show(); // display the current state for switch's



            }
        });


        textView.setText(arrayList.get(position).getTextView1());
        imageView.setImageResource(arrayList.get(position).getImageView1());



        return  rowView;
    }
}

    }

    }

        this.imageView1 = imageView1;
    }
}
